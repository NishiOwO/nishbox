152int gf_draw_platform_step(gf_draw_t* draw) {
153 int ret = 0;
154 glXMakeCurrent(draw->platform->display, draw->platform->window, draw->platform->context);
155 while(XPending(draw->platform->display) > 0) {
156 XEvent event;
157 XNextEvent(draw->platform->display, &event);
158 if(event.type == Expose) {
159 break;
160 } else if(event.type == ConfigureNotify) {
161 draw->x = event.xconfigure.x;
162 draw->y = event.xconfigure.y;
163 draw->width = event.xconfigure.width;
164 draw->height = event.xconfigure.height;
165 glXMakeCurrent(draw->platform->display, draw->platform->window, draw->platform->context);
166 gf_draw_reshape(draw);
167 } else if(event.type == ClientMessage) {
168 if(event.xclient.data.l[0] == draw->platform->wm_delete_window) {
169 draw->close = 1;
170 break;
171 }
172 }
173 }
174 if(ret == 0) {
175 gf_draw_driver_before(draw);
176 gf_draw_frame(draw);
177 gf_draw_driver_after(draw);
178
179 glXSwapBuffers(draw->platform->display, draw->platform->window);
180 }
181 return ret;
182}
